class GaussNB:
    def fit(self, X, y, epsilon=1e-3):
        self.likelihoods = dict()
        self.priors = dict()
        self.K = np.unique(y)  # Set of unique classes

        for k in self.K:
            X_k = X[y == k]
            # Naive Assumption: Observations are linearly independent of each other
            self.likelihoods[k] = {
                "mean": X_k.mean(axis=0),
                "var": X_k.var(axis=0) + epsilon  # Variance with smoothing
            }
            self.priors[k] = len(X_k) / len(X)

    def predict(self, X):
        N, D = X.shape
        P_hat = np.zeros((N, len(self.K)))

        for k in self.K:
            mean = self.likelihoods[k]["mean"]
            var = self.likelihoods[k]["var"]
            # Calculate the log probability using univariate Gaussian formula
            P_hat[:, k] = np.sum(-0.5 * np.log(2. * np.pi * var) - 0.5 * ((X - mean) ** 2) / var, axis=1) + np.log(self.priors[k])

        return self.K[P_hat.argmax(axis=1)]

class GaussBayes():
  def fit(self, X, y, epsilon=1e-2):
    self.likelihoods=dict()
    self.priors=dict()
    self.K=set(y.astype(int))

    for k in self.K:
      X_k = X[y==k, :]
      N_k, D = X_k.shape
      mu_k=X_k.mean(axis=0)

      self.likelihoods[k]={"mean":X_k.mean(axis=0), "cov":(1/(N_k-1))*np.matmul((X_k-mu_k).T, X_k-mu_k)+epsilon*np.identity(D)}
      self.priors[k]=len(X_k)/len(X)

  def predict(self,X):
    N, D = X.shape
    P_hat= np.zeros((N,len(self.K)))

    for k, l in self.likelihoods.items():
      P_hat[:,k]= mvn.logpdf(X, l["mean"],l["cov"])+np.log(self.priors[k])

    return P_hat.argmax(axis=1)

# Vectorized KNN Implementation
def knn_predict(X_train, y_train, X_test, k=3):
    num_test = X_test.shape[0]
    y_pred = np.zeros(num_test)
    
    for i in range(num_test):
        # Compute the Euclidean distance from the test point to all training points
        distances = np.sqrt(np.sum((X_train - X_test[i, :]) ** 2, axis=1))
        
        # Sort by distance and get the indices of the closest k points
        nearest_neighbors = np.argsort(distances)[:k]
        
        # Extract the labels of the k nearest points
        k_labels = y_train[nearest_neighbors]
        
        # Predict the most common label
        y_pred[i] = np.argmax(np.bincount(k_labels))
    
    return y_pred

import numpy as np

class KNN:
    def __init__(self, k=3, weighted=False):
        self.k = k
        self.weighted = weighted

    def fit(self, X, y):
        self.X_train = X
        self.y_train = y

    def predict(self, X):
        N, D = X.shape
        y_pred = np.zeros(N)
        
        for i in range(N):
            # Compute the Euclidean distance from the i-th test point to all training points
            distances = np.sqrt(np.sum((self.X_train - X[i, :]) ** 2, axis=1))
            
            # Get the indices of the k nearest neighbors
            nearest_neighbors = np.argsort(distances)[:self.k]
            
            # Extract the labels of the k nearest neighbors
            k_labels = self.y_train[nearest_neighbors]
            
            if self.weighted:
                # Compute weights as the inverse of distance
                weights = 1 / (distances[nearest_neighbors] + 1e-5)
                weighted_votes = np.zeros(len(np.unique(self.y_train)))
                for j, label in enumerate(k_labels):
                    weighted_votes[label] += weights[j]
                y_pred[i] = np.argmax(weighted_votes)
            else:
                # Predict the most common label
                y_pred[i] = np.argmax(np.bincount(k_labels))
        
        return y_pred

def normalize(X):
    return (X - np.mean(X, axis=0)) / np.std(X, axis=0)
